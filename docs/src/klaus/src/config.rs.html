#![allow(dead_code)]

use url::Url;
use toml;

use lsio::error::{Error, Result};
use lsio::config::{ConfigFile, ParseInto};

/// Config by default is located at /etc/lsio/config for a given server. You can pass in an option
/// on the cli ```-c "<whatever path you want>"``` and it will override the default.
///
/// If for some reason there is no config file and nothing is passed in the all of the
/// fields will be None for Option values or whatever the defaults are for a given type.
///
///
#[derive(Clone, Debug, PartialEq, Eq)]
pub struct Config {
    /// IP Address of server - defaults to 127.0.0.1
    pub ip: String,
    /// Port of server to use - defaults to 80
    pub port: u16,

    /// Section is for making outbound S3 calls.
    /// endpoint is in the format <scheme>://<fqdn>:<port>
    pub endpoint: Option<Url>,
    /// proxy is in the format <scheme>://<fqdn>:<port>
    pub proxy: Option<Url>,
    /// signature is either V2 or V4
    pub signature: String,
}

impl ConfigFile for Config {
    type Error = Error;

    fn from_toml(toml: toml::Value) -> Result<Self> {
        let mut cfg = Config::default();

        try!(toml.parse_into("options.ip", &mut cfg.ip));
        try!(toml.parse_into("options.port", &mut cfg.port));

        try!(toml.parse_into("outbound.endpoint", &mut cfg.endpoint));
        try!(toml.parse_into("outbound.proxy", &mut cfg.proxy));
        try!(toml.parse_into("outbound.signature", &mut cfg.signature));

        Ok(cfg)
    }
}

impl Default for Config {
    fn default() -> Self {
        Config {
            ip: "127.0.0.1".to_string(),
            port: 80,
            endpoint: None,
            proxy: None,
            signature: "V4".to_string(),
        }
    }
}

impl Config {
    pub fn set_endpoint(&mut self, value: Option<Url>) {
        self.endpoint = value;
    }

    pub fn set_proxy(&mut self, value: Option<Url>) {
        self.proxy = value;
    }

    pub fn set_signature(&mut self, value: String) {
        self.signature = value;
    }

    pub fn endpoint(&self) -> &Option<Url> {
        &self.endpoint
    }

    pub fn proxy(&self) -> &Option<Url> {
        &self.proxy
    }
}
